### HTML

```html
<div id="app">
  <nav>
    <span>Language：</span>
    <button @click="changeLan('en')">英</button>
    <button @click="changeLan('tw')">繁</button>
    <button @click="changeLan('cn')">簡</button>
    <button @click="changeLan('jp')">日</button>
  </nav>
  <h1 v-html="$t('message.hello')"></h1>
  <h1 v-html="$t('message.name')"></h1>  
</div>
```

### JavaScript

```js
const messages = {
  en: {
    message: {
      hello: 'Hello',
      name:'Mike'
    }
  },
  tw: {
    message: {
      hello: '哈囉',
      name:'麥克'
    }
  },
  cn: {
    message: {
      hello: '哈啰',
      name:'麦克'
    }
  },
  jp: {
    message: {
      hello: 'ハロー',
      name:'マイク'
    }
  }
}

let i18n = new VueI18n({
  locale: 'tw', //預設繁中
  messages,
})

let vm = new Vue({
  el: '#app',
  i18n,  
  data:{
    
  },
  methods:{
    changeLan(lan){
      i18n.locale = lan;
      console.log(i18n.locale);
    }
  }
})




```
